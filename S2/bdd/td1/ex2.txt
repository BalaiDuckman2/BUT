VAR FOURNISSEUR REAL RELATION{
    NO_FOURNISSEUR INTEGER
    capacite_centre INTEGER
    nom_pays INTEGER
}KEY{NO_FOURNISSEUR}

VAR ARTICLE REAL RELATION{
    NO_ARTICLE INTEGER
    nom_article CHAR
    couleur CHAR
    prix INTEGER
}KEY{NO_ARTICLE}

VAR PROJET REAL RELATION{
    NO_PROJET INTEGER
    nom_projet CHAR
    budget_projet CHAR
    ville_projet CHAR
}KEY{NO_PROJET}

VAR DOTATION REAL RELATION{
    quantite_dot
    NO_PROJET INTEGER
    NO_ARTICLE INTEGER
    NO_FOURNISSEUR INTEGER
}KEY{NO_PROJET,NO_ARTICLE,NO_FOURNISSEUR}

CONSTRAINT DOTATION_fk_FOURNISSEUR
    DOTATION{NO_FOURNISSEUR} <= FOURNISSEUR{NO_FOURNISSEUR}

CONSTRAINT DOTATION_fk_ARTICLE
    DOTATION{NO_ARTICLE} <= ARTICLE{NO_ARTICLE}

CONSTRAINT DOTATION_fk_PROJET
    DOTATION{NO_PROJET} <= PROJET{NO_PROJET}
